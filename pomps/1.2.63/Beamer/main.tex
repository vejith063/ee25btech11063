\begin{frame}{Question}
\text{Quadrature formula:} $\int_{0}^{1} f(x) \, dx \approx \frac{1}{12} [f(0) + 2bf(0.25) + 2f(0.5) + 2df(0.75) + f(1)]$. \\
If used as Simpson's $\frac{1}{3}$ rule: \hfill (2007 XE) 
\noindent
\\ a) b = d = 1 \hfill b) b = d = 2 \hfill c) b = 2d = 1 \hfill d) b = 2d = 2
\end{frame}
\begin{frame}{Simpson$'$s rule}
    Simpson's $1/3$ rule is a numerical integration method that approximates the area under a curve by fitting parabolic arcs (second-order polynomials) through sets of three points. 

For a single interval $[x_0, x_2]$ with a midpoint $x_1$ and step size $h$:
\begin{equation}
\int_{x_0}^{x_2} f(x) \, dx \approx \frac{h}{3} \left[ f(x_0) + 4f(x_1) + f(x_2) \right]
\end{equation}

In general \text{Simpson's 1/3 Rule} over $n$ intervals (where $n$ must be even):
\begin{equation}
\int_{a}^{b} f(x) \, dx \approx \frac{h}{3} \left[ f(x_0) + 4 \sum_{i=1,3,5}^{n-1} f(x_i) + 2 \sum_{j=2,4,6}^{n-2} f(x_j) + f(x_n) \right]
\end{equation}
Where $h = \frac{b-a}{n}$ is the uniform step size and $x_0$ = a, $x_1$ = b.
\end{frame}
\begin{frame}{Solution}
    The problem provides a specific quadrature formula:
\begin{equation}
\int_{0}^{1} f(x) \, dx \approx \frac{1}{12} \left[ f(0) + 2bf(0.25) + 2f(0.5) + 2df(0.75) + f(1) \right]
\label{eq:given}
\end{equation}

Based on the function arguments $\{0, 0.25, 0.5, 0.75, 1\}$:
\begin{itemize}
    \item $a = 0, b = 1$
    \item $n = 4$ intervals
    \item $h = \frac{1 - 0}{4} = 0.25$
\end{itemize}

Substituting $h = 0.25$ into the general Simpson's Rule for $n=4$:
\begin{equation}
\int_{0}^{1} f(x) \, dx \approx \frac{0.25}{3} \left[ f(0) + 4f(0.25) + 2f(0.5) + 4f(0.75) + f(1) \right]
\end{equation}

Simplifying  $\frac{0.25}{3} = \frac{1}{12}$, the standard internal weights are:
\begin{equation}
\text{Internal Weights} = [4, 2, 4]
\end{equation}
\end{frame}
\begin{frame}{Solution}
    Comparing these to the coefficients inside the brackets of the equation (3) we get
\begin{itemize}
    \item Coefficient for $f(0.25)$: $2b = 4 \implies \mathbf{b = 2}$
    \item Coefficient for $f(0.75)$: $2d = 4 \implies \mathbf{d = 2}$
\end{itemize}
\textbf{Numerical Example}\\
Lets test the above relation with $f(x) = x$
\\ \textbf{ Exact Integral :}\\
\begin{equation}
\int_{0}^{1} x \, dx = \left[ \frac{x^2}{2} \right]_0^1 = \frac{1}{2} = 0.5
\end{equation}
\textbf{ Applying the Formula} \\
\begin{equation}
\int_{0}^{1} f(x) \, dx \approx \frac{1}{12} \left[ f(0) + 4f(0.25) + 2f(0.5) + 4f(0.75) + f(1) \right]
\end{equation}
\end{frame}
\begin{frame}{Solution}
    \begin{itemize}
    \item $f(0) = 0$
    \item $f(0.25) = 0.25 = \frac{1}{4}$
    \item $f(0.5) = 0.5 = \frac{1}{2}$
    \item $f(0.75) = 0.75 = \frac{3}{4}$
    \item $f(1) = 1$
\end{itemize}
Substitute these into above equation  and simplifying  it will give :
\begin{align}
\frac{1}{12} \left[ 0 + 4\left(\frac{1}{4}\right) + 2\left(\frac{1}{2}\right) + 4\left(\frac{3}{4}\right) + 1 \right]  \\
\implies \frac{1}{12} \left[ 1 + 1 + 3 + 1 \right] \\
\implies \frac{6}{12} = \frac{1}{2}
\end{align}
\end{frame}

import numpy as np
import matplotlib.pyplot as plt

def f(x):
    return x

a = 0.0
b = 1.0
n = 4

# Generate the nodes
x_nodes = np.linspace(a, b, n + 1)
y_nodes = f(x_nodes)

x_smooth = np.linspace(a, b, 200)
y_smooth = f(x_smooth)

plt.figure(figsize=(10, 6))
\end{lstlisting}

\begin{frame}[fragile]{Python: plot.py}
\begin{lstlisting}[language=Python]
plt.plot(x_smooth, y_smooth, 'k-', label='Original Function: f(x) = x', linewidth=3, alpha=0.4)

#  Implement Simpson's 1/3 Rule 
for i in range(0, n, 2):
    # Get the 3 points (x0, x1, x2) for this parabolic block
    x_triplet = x_nodes[i:i+3]
    y_triplet = y_nodes[i:i+3]
    
    # Fit a 2nd-degree polynomial (a parabola) to these 3 points
    parabola_coeffs = np.polyfit(x_triplet, y_triplet, 2)
    parabola_func = np.poly1d(parabola_coeffs)
    \end{lstlisting}
